\makeatletter

  \begin{svitemize}[#1]%
  \let\olditem\item
  \renewcommand\item[1][\itemsymbol]{
    \ifnum\@itemdepth=1\relax
      \vfill
    \else
      \ifnum\@itemdepth=2\relax
        \setlength\itemsep{0.5ex}
      \else
        \ifnum\@itemdepth=3\relax
          \setlength\itemsep{0.25ex}
    \fi\fi\fi
    \svitem[##1]
  }
}{
  \end{svitemize}\vfill
}

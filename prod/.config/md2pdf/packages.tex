% File containing all packages included in the md2pdf-framework

% Setup encoding
\usepackage[utf8]{inputenc}

% Allow if/then/else with extended commands
\usepackage{ifthen}
\usepackage{xifthen}

% Allow setting fonts
\usepackage{fontspec}

% Allow usage of colored boxes
\usepackage{tcolorbox}
\tcbuselibrary{breakable}
\tcbuselibrary{skins}

% Allow checking if page is in draft mode
\usepackage{ifdraft}

% Allow using pictures 
\usepackage{graphicx}

% Allow configuring header
\usepackage{fancyhdr}

% Allow extra mark commands 
\usepackage{extramarks}

% Get page-number of last page
\usepackage{lastpage}

% Allow advanced string modifications
\usepackage{xstring}
